reporter_run(ProblemData pd):
    ReportEntryList pel = lr_easy_search($pd)

    if $pel is empty
        ReportEntry re = lr_easy_report($pd)
        add $re to $pel

    foreach ReportEntry p in $pel:
        append "$p.reporter.name: URL=$p.url" to the problem data's reported_to

    show_reports(pel)


lr_easy_report(ProblemData pd):
    Report r = lr_easy_build_report($pd)
    ReportEntry re = lr_easy_report_create($r)
    return $re
